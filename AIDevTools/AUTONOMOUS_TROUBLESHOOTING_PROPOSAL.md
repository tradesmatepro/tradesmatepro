```javascript
// AIDevTools/supabaseManagementAPI.js

const SUPABASE_ACCESS_TOKEN = process.env.SUPABASE_ACCESS_TOKEN;
const SUPABASE_PROJECT_REF = process.env.SUPABASE_PROJECT_REF;

class SupabaseManagementAPI {
  // Get Edge Function logs
  async getEdgeFunctionLogs(functionName, limit = 100) {
    const response = await fetch(
      `https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/functions/${functionName}/logs?limit=${limit}`,
      {
        headers: {
          'Authorization': `Bearer ${SUPABASE_ACCESS_TOKEN}`,
          'Content-Type': 'application/json'
        }
      }
    );
    return response.json();
  }

  // Update secret
  async updateSecret(name, value) {
    const response = await fetch(
      `https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/secrets`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${SUPABASE_ACCESS_TOKEN}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ name, value })
      }
    );
    return response.json();
  }

  // Deploy Edge Function
  async deployFunction(functionName) {
    const response = await fetch(
      `https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/functions/${functionName}/deploy`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${SUPABASE_ACCESS_TOKEN}`,
          'Content-Type': 'application/json'
        }
      }
    );
    return response.json();
  }

  // List all secrets
  async listSecrets() {
    const response = await fetch(
      `https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/secrets`,
      {
        headers: {
          'Authorization': `Bearer ${SUPABASE_ACCESS_TOKEN}`,
          'Content-Type': 'application/json'
        }
      }
    );
    return response.json();
  }
}

module.exports = new SupabaseManagementAPI();
```

#### Resend API Integration

```javascript
// AIDevTools/resendAPI.js

const RESEND_API_KEY = process.env.RESEND_API_KEY;

class ResendAPI {
  // Validate API key
  async validateAPIKey(apiKey = RESEND_API_KEY) {
    try {
      const response = await fetch('https://api.resend.com/api-keys', {
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json'
        }
      });
      return response.ok;
    } catch (error) {
      return false;
    }
  }

  // Create new API key
  async createAPIKey(name, permission = 'full_access') {
    const response = await fetch('https://api.resend.com/api-keys', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${RESEND_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ name, permission })
    });
    return response.json();
  }

  // List API keys
  async listAPIKeys() {
    const response = await fetch('https://api.resend.com/api-keys', {
      headers: {
        'Authorization': `Bearer ${RESEND_API_KEY}`,
        'Content-Type': 'application/json'
      }
    });
    return response.json();
  }

  // Test email sending
  async sendTestEmail(to, from, subject, body) {
    const response = await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${RESEND_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        from,
        to,
        subject,
        html: body
      })
    });
    return response.json();
  }
}

module.exports = new ResendAPI();
```

```javascript
// AIDevTools/autoFixResendAPIKey.js

const supabaseAPI = require('./supabaseManagementAPI');
const resendAPI = require('./resendAPI');
const { appendToPhaseLog } = require('./phaseLogger');

async function autoFixResendAPIKey() {
  console.log('üîß Starting autonomous fix for Resend API key...\n');

  // Step 1: Validate current key
  console.log('üìã Step 1: Validating current Resend API key...');
  const currentKeyValid = await resendAPI.validateAPIKey();
  
  if (currentKeyValid) {
    console.log('   ‚úÖ Current API key is valid - no fix needed');
    return { success: true, message: 'API key already valid' };
  }
  
  console.log('   ‚ùå Current API key is invalid');
  appendToPhaseLog('üîç Detected invalid Resend API key');

  // Step 2: Create new API key
  console.log('\nüìã Step 2: Creating new Resend API key...');
  const newKey = await resendAPI.createAPIKey('TradeMate-Pro-Auto-Generated');
  
  if (!newKey.id) {
    console.log('   ‚ùå Failed to create new API key');
    return { success: false, error: 'Failed to create API key' };
  }
  
  console.log(`   ‚úÖ New API key created: ${newKey.id}`);
  appendToPhaseLog(`‚úÖ Created new Resend API key: ${newKey.id}`);

  // Step 3: Update Supabase secret
  console.log('\nüìã Step 3: Updating Supabase secret...');
  const updateResult = await supabaseAPI.updateSecret('RESEND_API_KEY', newKey.token);
  
  if (!updateResult.success) {
    console.log('   ‚ùå Failed to update Supabase secret');
    return { success: false, error: 'Failed to update secret' };
  }
  
  console.log('   ‚úÖ Supabase secret updated');
  appendToPhaseLog('‚úÖ Updated RESEND_API_KEY in Supabase secrets');

  // Step 4: Redeploy Edge Function
  console.log('\nüìã Step 4: Redeploying Edge Function...');
  const deployResult = await supabaseAPI.deployFunction('send-quote-email');
  
  if (!deployResult.success) {
    console.log('   ‚ùå Failed to redeploy Edge Function');
    return { success: false, error: 'Failed to redeploy function' };
  }
  
  console.log('   ‚úÖ Edge Function redeployed');
  appendToPhaseLog('‚úÖ Redeployed send-quote-email Edge Function');

  // Step 5: Test email sending
  console.log('\nüìã Step 5: Testing email sending...');
  const testResult = await resendAPI.sendTestEmail(
    'test@example.com',
    'noreply@trademateapp.com',
    'Test Email - Auto Fix',
    '<p>This is a test email to verify the API key is working.</p>'
  );
  
  if (testResult.id) {
    console.log(`   ‚úÖ Test email sent successfully: ${testResult.id}`);
    appendToPhaseLog(`‚úÖ Test email sent successfully: ${testResult.id}`);
  } else {
    console.log('   ‚ö†Ô∏è  Test email failed, but API key should be valid');
  }

  console.log('\nüéâ Autonomous fix complete!');
  appendToPhaseLog('üéâ Autonomous fix for Resend API key complete');

  return {
    success: true,
    newKeyId: newKey.id,
    testEmailId: testResult.id
  };
}

// CLI support
if (require.main === module) {
  autoFixResendAPIKey()
    .then(result => {
      console.log('\nüìä Result:', result);
      process.exit(result.success ? 0 : 1);
    })
    .catch(error => {
      console.error('\n‚ùå Error:', error);
      process.exit(1);
    });
}

module.exports = { autoFixResendAPIKey };
```

```javascript
  await page.fill('input[name="name"]', name);
```

**Pros:**
- Can handle complex UI flows
- Works even if API is not available
- Can verify changes visually

**Cons:**
- Requires storing credentials
- May break if UI changes
- Slower than API calls
- May trigger security alerts
